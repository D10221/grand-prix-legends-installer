Grand Prix Legends wine Installer v0.0.1
Usage: 'gplnx [verb] [options]'
Verbs: 
  install [options?] [what]?   'run GPL installers'
      options:
          no options implemented
      what:
         values: gpl gem vbr
      example:
          $gpli install -r gpl # REINSTALL gpl
          $gpli install -r gpl gem vbr # reinstall gpl gem vbr
  uninstall [options?] 'remove GPL,GEM and wine Prefix'
      options: 
          'gpl'   remove 'c:\Sierra\gpl'
          'geml'  remove 'c:\Sierra\gpl'
          'pfx'      remove 'prefix' (default)
  run                   "run something in gpl's WINEPREFIX"
      example:    '$gplnx run wine winecfg'
  wine                  "run wine 'something' gpl's WINEPREFIX"
      example:    '$gplnx wine winecfg'
  open
      example: '$gplnx open' # opens $WINEPREFIX
      example: '$gplnx open dirve_c ' # opens $WINEPREFIX/drive_c
  launch
      example: '$gplnx launch gpl'  # starts wine 'c:\Sierra\GPL\gpl.exe'
      example: '$gplnx launch gem+' # starts wine 'c:\GPLSecrets\GEM+\GEMP2.exe'

Options:
  -h --help      'Show this'
  -v --version   'Show 'gplnx' version'
  --nomount      'Do not try to mount ISO'

Vars:
 NOMOUNT         'Do not try to mount ISO'
 NOUSERMOUNT     'Do not try to mount ISO in userspace'
 NOROOTMOUNT     'Do not try to mount ISO if requires root access
 DEBUG           'show debug info'
                  example '$ DEBUG='*' ./gpli.local install'
